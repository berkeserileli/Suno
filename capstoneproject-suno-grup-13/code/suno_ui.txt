# TCP sunucu dinleme iÅŸlemi arka planda
class TCPServer(QThread):
    veri_alindi = pyqtSignal(dict)

    def run(self):
        HOST = '0.0.0.0'
        PORT = 5005

        server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        server.bind((HOST, PORT))
        server.listen(5)
        print("âœ… ArayÃ¼z sunucu gibi Ã§alÄ±ÅŸÄ±yor ve dinlemede...")

        while True:
            conn, addr = server.accept()
            print("ğŸ”— BaÄŸlantÄ± geldi:", addr)
            try:
                data = conn.recv(4096).decode("utf-8")
                print("ğŸ“© Gelen veri:", data)
                parsed = json.loads(data)
                self.veri_alindi.emit(parsed)
            except Exception as e:
                print("âŒ JSON HatasÄ±:", e)
            finally:
                conn.close()

    def guncelle(self, veri):
